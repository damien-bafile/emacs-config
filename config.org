  (use-package persp-mode
    :config (persp-mode +1)
    :custom
    (persp-autokill-buffer-on-remove 'kill-weak)
    (persp-add buffer-predicate nil)
    (persp-auto-save-opt 0)
    (persp-auto-save-persp-name "main")
    (persp-save-dir (expand-file-name "persp-confs/" user-emacs-directory))
    (persp-modestring-use-layouts-name t)
    :config
    ;; Don't kill scratch buffer when switching workspaces
    (add-hook 'persp-after-switch-functions
              (lambda (&rest _)
                (when (get-buffer "*scratch*")
                  (persp-add-buffer "*scratch*")))))

  ;; Doom-style workspace helpers built on persp-mode
  (unless (fboundp 'persp-list)
    (defun persp-list ()
      "Return a list of all perspective objects."
      (cl-remove-if-not #'perspective-p
                        (mapcar #'persp-get-by-name (persp-names)))))

  (defun my/workspace-display ()
    "Display open workspaces in the echo area."
    (interactive)
    (let* ((names (mapcar #'persp-name (persp-list)))
           (current (safe-persp-name (get-current-persp)))
           (formatted (mapcar (lambda (name)
                                (if (string= name current)
                                    (propertize (format "[%s]" name) 'face 'bold)
                                  name))
                              names)))
      (message "Workspaces: %s" (string-join formatted " "))))

  (defun my/workspace-new ()
    "Create a new blank workspace."
    (interactive)
    (let ((name (format "#%d" (1+ (length (persp-list))))))
      (persp-switch name)
      (my/workspace-display)))

  (defun my/workspace-new-named (name)
    "Create a new workspace with NAME."
    (interactive "sWorkspace name: ")
    (persp-switch name)
    (my/workspace-display))

  (defun my/workspace-switch-to (index)
    "Switch to workspace at INDEX (0-based)."
    (interactive "nWorkspace index: ")
    (let ((persps (persp-list)))
      (when (< index (length persps))
        (persp-switch (persp-name (nth index persps))))))

  (defun my/workspace-switch-by-name ()
    "Switch to a workspace by name via completing-read."
    (interactive)
    (let ((names (mapcar #'persp-name (persp-list))))
      (when names
        (persp-switch (completing-read "Workspace: " names nil t)))))

  (defun my/workspace-switch-final ()
    "Switch to the last workspace."
    (interactive)
    (let ((persps (persp-list)))
      (when persps
        (persp-switch (persp-name (car (last persps)))))))

  (defun my/workspace-cycle (direction)
    "Cycle workspaces in DIRECTION (+1 forward, -1 backward)."
    (interactive "p")
    (let* ((persps (persp-list))
           (current-index (cl-position (get-current-persp) :test #'eq))
           (next-index (mod (+ current-index direction) (length persps))))
      (persp-switch (persp-name (nth next-index persps)))))

  (defun my/workspace-kill ()
    "Kill the current workspace (switch to first if last)."
    (interactive)
    (when (> (length (persp-list)) 1)
      (persp-kill (safe-persp-name (get-current-persp)))))

  (defun my/workspace-rename ()
    "Rename the current workspace."
    (interactive)
    (let ((new-name (read-string "New name: " (safe-persp-name (get-current-persp)))))
      (persp-rename new-name (get-current-persp))
      (my/workspace-display)))

  (defun my/workspace-move-buffer-left ()
    "Move current buffer to the previous workspace."
    (interactive)
    (let* ((persps (persp-list))
           (current-index (cl-position (get-current-persp) :test #'eq))
           (target-index (mod (1- current-index) (length persps)))
           (target (nth target-index persps)))
      (persp-remove-buffer (current-buffer) (get-current-persp) nil nil)
      (persp-add-buffer (current-buffer) target nil nil)))

  (defun my/workspace-move-buffer-right ()
    "Move current buffer to the next workspace."
    (interactive)
    (let* ((persps (persp-list))
           (current-index (cl-position (get-current-persp) :test #'eq))
           (target-index (mod (1+ current-index) (length persps)))
           (target (nth target-index persps)))
      (persp-remove-buffer (current-buffer) (get-current-persp) nil nil)
      (persp-add-buffer (current-buffer) target nil nil)))

  (defun my/workspace-save ()
    "Save the current workspace session to file."
    (interactive)
    (persp-save-state-to-file)
    (message "Workspace saved"))

  (defun my/workspace-load ()
    "Load a saved workspace session from file."
    (interactive)
    (persp-load-state-from-file))

  (defun my/workspace-restore-last ()
    "Restore the last saved session."
    (interactive)
    (persp-load-state-from-file))

Python/uv:  prompt for uv run command

Leiningen: lein run"
  (interactive)

      (compile "just deploy"))

      (compile "cargo run"))
     ((and root (file-exists-p (expand-file-name "project.clj" root)))
      (compile "lein run"))
     ((and root (file-exists-p (expand-file-name "uv.lock" root)))
      (compile (read-shell-command "Run: " "uv run python main.py")))
     ((and root (file-exists-p (expand-file-name "CMakeLists.txt" root)))
      (compile "just run"))
